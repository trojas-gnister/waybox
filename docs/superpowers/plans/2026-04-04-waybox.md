# Waybox Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Rust CLI tool that creates NixOS VMs and displays their applications as native Wayland windows on the host desktop via waypipe over vsock.

**Architecture:** WayboxConfig is the single source of truth — it drives both libvirt domain XML generation and NixOS guest configuration via Askama templates. All guest communication uses vsock (no SSH). The `virt` crate provides native libvirt bindings, eliminating virsh shell-outs.

**Tech Stack:** Rust, clap 4.5, serde + toml, thiserror, askama 0.15, virt 0.4 (libvirt bindings), rand, log + env_logger

**Design spec:** `docs/superpowers/specs/2026-04-04-waybox-design.md`

```rust
    #[error("VM '{0}' already exists")]
    VmAlreadyExists(String),

    // Libvirt
    #[error("Libvirt error: {0}")]
    Libvirt(#[from] virt::error::Error),

    #[error("VM '{name}' not found in libvirt")]
    VmNotFound { name: String },

    #[error("VM '{name}' is already {state}")]
    VmWrongState { name: String, state: String },

    // NixOS
    #[error("NixOS image build failed: {0}")]
    ImageBuild(String),

    #[error("Prerequisite not found: {tool}. {hint}")]
    PrerequisiteNotFound { tool: String, hint: String },

    // Display
    #[error("waypipe session failed: {0}")]
    Waypipe(String),

    #[error("Audio bridge failed: {0}")]
    Audio(String),

    #[error("Guest not responding on vsock CID {cid}")]
    VsockTimeout { cid: u32 },

    // USB
    #[error("USB device {id} not found on host")]
    UsbNotFound { id: String },

    // Generic
    #[error("IO error ({context}): {source}")]
    Io {
        context: String,
        source: std::io::Error,
    },

    #[error("TOML serialization error: {0}")]
    TomlSerialize(#[from] toml::ser::Error),

    #[error("TOML deserialization error: {0}")]
    TomlDeserialize(#[from] toml::de::Error),

    #[error("Template rendering error: {0}")]
    Template(#[from] askama::Error),
}

/// Convenience trait for adding IO error context
pub trait IoContext<T> {
    fn io_context(self, context: &str) -> Result<T, WayboxError>;
}

impl<T> IoContext<T> for Result<T, std::io::Error> {
    fn io_context(self, context: &str) -> Result<T, WayboxError> {
        self.map_err(|source| WayboxError::Io {
            context: context.to_string(),
            source,
        })
    }
}

pub type Result<T> = std::result::Result<T, WayboxError>;
```

```rust
use crate::error::{Result, WayboxError};

pub fn validate_vm_name(name: &str) -> Result<()> {
    if name.is_empty() {
        return Err(WayboxError::InvalidName {
            name: name.to_string(),
            reason: "name cannot be empty".to_string(),
        });
    }
    if name.len() > 64 {
        return Err(WayboxError::InvalidName {
            name: name.to_string(),
            reason: "name cannot exceed 64 characters".to_string(),
        });
    }
    if !name.starts_with(|c: char| c.is_ascii_alphabetic()) {
        return Err(WayboxError::InvalidName {
            name: name.to_string(),
            reason: "name must start with a letter".to_string(),
        });
    }
    if !name.chars().all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_') {
        return Err(WayboxError::InvalidName {
            name: name.to_string(),
            reason: "name may only contain letters, digits, hyphens, and underscores".to_string(),
        });
    }
    Ok(())
}

pub fn validate_usb_id(id: &str) -> Result<()> {
    let parts: Vec<&str> = id.split(':').collect();
    if parts.len() != 2 {
        return Err(WayboxError::InvalidUsbId { id: id.to_string() });
    }
    let (vendor, product) = (parts[0], parts[1]);
    if vendor.len() != 4
        || product.len() != 4
        || !vendor.chars().all(|c| c.is_ascii_hexdigit())
        || !product.chars().all(|c| c.is_ascii_hexdigit())
    {
        return Err(WayboxError::InvalidUsbId { id: id.to_string() });
    }
    Ok(())
}

pub fn validate_share_path(share: &str) -> Result<(String, String)> {
    if share.is_empty() {
        return Err(WayboxError::InvalidSharePath {
            path: share.to_string(),
            reason: "share path cannot be empty".to_string(),
        });
    }
    let Some((host, guest)) = share.split_once(':') else {
        return Err(WayboxError::InvalidSharePath {
            path: share.to_string(),
            reason: "expected format host_path:guest_path".to_string(),
        });
    };
    if !host.starts_with('/') {
        return Err(WayboxError::InvalidSharePath {
            path: share.to_string(),
            reason: "host path must be absolute".to_string(),
        });
    }
    if !guest.starts_with('/') {
        return Err(WayboxError::InvalidSharePath {
            path: share.to_string(),
            reason: "guest path must be absolute".to_string(),
        });
    }
    Ok((host.to_string(), guest.to_string()))
}
```

```rust
/// Network mode for the VM
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub enum NetworkMode {
    Nat,
    Airgapped,
}

impl Default for NetworkMode {
    fn default() -> Self {
        NetworkMode::Nat
    }
}

/// A USB device identified by vendor:product ID
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub struct UsbDevice {
    pub vendor: String,
    pub product: String,
}

impl UsbDevice {
    pub fn from_id(id: &str) -> crate::error::Result<Self> {
        validation::validate_usb_id(id)?;
        let (vendor, product) = id.split_once(':').unwrap();
        Ok(Self {
            vendor: vendor.to_lowercase(),
            product: product.to_lowercase(),
        })
    }

    pub fn id(&self) -> String {
        format!("{}:{}", self.vendor, self.product)
    }
}

/// A shared folder between host and guest
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub struct SharedFolder {
    pub host_path: String,
    pub guest_path: String,
}

/// vsock port assignments (fixed per VM, CID differentiates VMs)
pub const VSOCK_DISPLAY_PORT: u32 = 5000;
pub const VSOCK_AUDIO_PORT: u32 = 5001;
pub const VSOCK_CONTROL_PORT: u32 = 5002;

/// Minimum vsock CID (0=hypervisor, 1=reserved, 2=host)
pub const VSOCK_CID_MIN: u32 = 3;

/// Complete VM configuration — single source of truth
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub struct WayboxConfig {
    pub name: String,

    // Resources
    #[serde(default = "default_memory")]
    pub memory_mb: u32,
    #[serde(default = "default_vcpus")]
    pub vcpus: u32,
    #[serde(default = "default_disk")]
    pub disk_gb: u32,

    // Packages
    #[serde(default)]
    pub system_packages: Vec<String>,
    #[serde(default)]
    pub flatpak_packages: Vec<String>,

    // Devices
    #[serde(default)]
    pub usb_devices: Vec<UsbDevice>,
    #[serde(default)]
    pub shared_folders: Vec<SharedFolder>,

    // Network
    #[serde(default)]
    pub network_mode: NetworkMode,

    // Display
    pub vsock_cid: u32,

    // Flags
    #[serde(default)]
    pub headless: bool,
    #[serde(default)]
    pub share_readonly: bool,
}

fn default_memory() -> u32 { 2048 }
fn default_vcpus() -> u32 { 2 }
fn default_disk() -> u32 { 20 }

impl WayboxConfig {
    /// Validate the entire config
    pub fn validate(&self) -> crate::error::Result<()> {
        validation::validate_vm_name(&self.name)?;
        for usb in &self.usb_devices {
            validation::validate_usb_id(&usb.id())?;
        }
        Ok(())
    }

    /// Path to this VM's config file
    pub fn config_path(&self) -> std::path::PathBuf {
        Self::config_dir().join(format!("{}.toml", self.name))
    }

    /// Path to the config directory
    pub fn config_dir() -> std::path::PathBuf {
        dirs_path("config")
    }

    /// Path to this VM's disk image
    pub fn image_path(&self) -> std::path::PathBuf {
        Self::images_dir().join(format!("{}.qcow2", self.name))
    }

    /// Path to the images directory
    pub fn images_dir() -> std::path::PathBuf {
        dirs_path("data").join("images")
    }

    /// Save config to TOML file
    pub fn save(&self) -> crate::error::Result<()> {
        let dir = Self::config_dir();
        std::fs::create_dir_all(&dir).map_err(|e| crate::error::WayboxError::Io {
            context: format!("creating config dir {:?}", dir),
            source: e,
        })?;
        let toml_str = toml::to_string_pretty(self)?;
        let path = self.config_path();
        std::fs::write(&path, toml_str).map_err(|e| crate::error::WayboxError::Io {
            context: format!("writing config {:?}", path),
            source: e,
        })?;
        log::info!("Saved config to {:?}", path);
        Ok(())
    }

    /// Load config from TOML file
    pub fn load(name: &str) -> crate::error::Result<Self> {
        let path = Self::config_dir().join(format!("{name}.toml"));
        if !path.exists() {
            return Err(crate::error::WayboxError::ConfigNotFound(name.to_string()));
        }
        let content = std::fs::read_to_string(&path).map_err(|e| crate::error::WayboxError::Io {
            context: format!("reading config {:?}", path),
            source: e,
        })?;
        let config: Self = toml::from_str(&content)?;
        Ok(config)
    }

    /// List all saved VM configs
    pub fn list_all() -> crate::error::Result<Vec<Self>> {
        let dir = Self::config_dir();
        if !dir.exists() {
            return Ok(vec![]);
        }
        let mut configs = Vec::new();
        let entries = std::fs::read_dir(&dir).map_err(|e| crate::error::WayboxError::Io {
            context: format!("reading config dir {:?}", dir),
            source: e,
        })?;
        for entry in entries {
            let entry = entry.map_err(|e| crate::error::WayboxError::Io {
                context: "reading config dir entry".to_string(),
                source: e,
            })?;
            let path = entry.path();
            if path.extension().is_some_and(|ext| ext == "toml")
                && path.file_name().is_some_and(|n| n != "vm-passwords.toml")
            {
                let content = std::fs::read_to_string(&path).map_err(|e| {
                    crate::error::WayboxError::Io {
                        context: format!("reading {:?}", path),
                        source: e,
                    }
                })?;
                if let Ok(config) = toml::from_str::<Self>(&content) {
                    configs.push(config);
                }
            }
        }
        Ok(configs)
    }

    /// Delete config file
    pub fn delete(&self) -> crate::error::Result<()> {
        let path = self.config_path();
        if path.exists() {
            std::fs::remove_file(&path).map_err(|e| crate::error::WayboxError::Io {
                context: format!("deleting config {:?}", path),
                source: e,
            })?;
        }
        Ok(())
    }

    /// Find next available vsock CID by scanning existing configs
    pub fn next_available_cid() -> crate::error::Result<u32> {
        let configs = Self::list_all()?;
        let used_cids: std::collections::HashSet<u32> =
            configs.iter().map(|c| c.vsock_cid).collect();
        let mut cid = VSOCK_CID_MIN;
        while used_cids.contains(&cid) {
            cid += 1;
        }
        Ok(cid)
    }
}

/// Resolve ~/.config/waybox or ~/.local/share/waybox
fn dirs_path(kind: &str) -> std::path::PathBuf {
    let home = std::env::var("HOME").unwrap_or_else(|_| "/tmp".to_string());
    match kind {
        "config" => std::path::PathBuf::from(&home).join(".config").join("waybox"),
        "data" => std::path::PathBuf::from(&home)
            .join(".local")
            .join("share")
            .join("waybox"),
        _ => unreachable!(),
    }
}
```

```rust
use crate::config::{NetworkMode, WayboxConfig};

pub fn generate_domain_xml(config: &WayboxConfig) -> String {
    let mut xml = String::with_capacity(4096);

    xml.push_str(&format!(
        "<domain type='kvm'>
  <name>{name}</name>
  <memory unit='MiB'>{memory}</memory>
  <vcpu>{vcpus}</vcpu>
  <os>
    <type arch='x86_64' machine='q35'>hvm</type>
    <boot dev='hd'/>
  </os>
  <features>
    <acpi/>
    <apic/>
  </features>
  <devices>
    <emulator>/usr/bin/qemu-system-x86_64</emulator>
    <disk type='file' device='disk'>
      <driver name='qemu' type='qcow2'/>
      <source file='{image_path}'/>
      <target dev='vda' bus='virtio'/>
    </disk>",
        name = config.name,
        memory = config.memory_mb,
        vcpus = config.vcpus,
        image_path = config.image_path().display(),
    ));

    // Network interface (NAT only)
    if config.network_mode == NetworkMode::Nat {
        xml.push_str(
            "
    <interface type='network'>
      <source network='default'/>
      <model type='virtio'/>
    </interface>",
        );
    }

    // Serial console
    xml.push_str(
        "
    <serial type='pty'>
      <target port='0'/>
    </serial>
    <console type='pty'>
      <target type='serial' port='0'/>
    </console>",
    );

    // Venus/Vulkan GPU (non-headless only)
    if !config.headless {
        xml.push_str(
            "
    <video>
      <model type='virtio' heads='1' 3d='yes'/>
    </video>",
        );
    }

    // vsock
    xml.push_str(&format!(
        "
    <vsock model='virtio'>
      <cid auto='no' val='{cid}'/>
    </vsock>",
        cid = config.vsock_cid,
    ));

    // USB devices
    for usb in &config.usb_devices {
        xml.push_str(&format!(
            "
    <hostdev mode='subsystem' type='usb' managed='yes'>
      <source>
        <vendor id='0x{vendor}'/>
        <product id='0x{product}'/>
      </source>
    </hostdev>",
            vendor = usb.vendor,
            product = usb.product,
        ));
    }

    // Shared folders (virtiofs)
    for (i, share) in config.shared_folders.iter().enumerate() {
        xml.push_str(&format!(
            "
    <filesystem type='mount' accessmode='passthrough'>
      <driver type='virtiofs'/>
      <source dir='{host_path}'/>
      <target dir='fs{i}'/>{readonly}
    </filesystem>",
            host_path = share.host_path,
            i = i,
            readonly = if config.share_readonly {
                "\n      <readonly/>"
            } else {
                ""
            },
        ));
    }

    // Memory backing for virtiofs (needed if any shared folders)
    if !config.shared_folders.is_empty() {
        // Close devices, add memoryBacking, then re-handle
        xml.push_str(
            "
  </devices>
  <memoryBacking>
    <source type='memfd'/>
    <access mode='shared'/>
  </memoryBacking>
</domain>",
        );
    } else {
        xml.push_str(
            "
  </devices>
</domain>",
        );
    }

    xml
}
```

```rust
pub mod domain;

use crate::error::{Result, WayboxError};
use virt::connect::Connect;
use virt::domain::Domain;

/// Represents the state of a VM domain
#[derive(Debug, Clone, Copy, PartialEq)]
pub enum DomainState {
    Running,
    Shutoff,
    Paused,
    Shutdown,
    Other(u32),
}

impl From<u32> for DomainState {
    fn from(state: u32) -> Self {
        match state {
            1 => DomainState::Running,
            3 => DomainState::Paused,
            4 => DomainState::Shutdown,
            5 => DomainState::Shutoff,
            other => DomainState::Other(other),
        }
    }
}

impl std::fmt::Display for DomainState {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            DomainState::Running => write!(f, "running"),
            DomainState::Shutoff => write!(f, "shutoff"),
            DomainState::Paused => write!(f, "paused"),
            DomainState::Shutdown => write!(f, "shutting down"),
            DomainState::Other(n) => write!(f, "unknown state {n}"),
        }
    }
}

/// Summary info about a domain
pub struct DomainInfo {
    pub name: String,
    pub state: DomainState,
}

/// Wrapper around the virt crate's libvirt connection
pub struct LibvirtConnection {
    conn: Connect,
}

impl LibvirtConnection {
    /// Connect to qemu:///system
    pub fn new() -> Result<Self> {
        let conn = Connect::open(Some("qemu:///system"))?;
        Ok(Self { conn })
    }

    /// Define a VM from domain XML (persistent, not started)
    pub fn define_vm(&self, xml: &str) -> Result<()> {
        Domain::define_xml(&self.conn, xml)?;
        Ok(())
    }

    /// Start a defined VM
    pub fn start_vm(&self, name: &str) -> Result<()> {
        let domain = self.lookup(name)?;
        if domain.is_active()? {
            return Err(WayboxError::VmWrongState {
                name: name.to_string(),
                state: "running".to_string(),
            });
        }
        domain.create()?;
        Ok(())
    }

    /// Gracefully stop a running VM
    pub fn stop_vm(&self, name: &str) -> Result<()> {
        let domain = self.lookup(name)?;
        if !domain.is_active()? {
            return Err(WayboxError::VmWrongState {
                name: name.to_string(),
                state: "shutoff".to_string(),
            });
        }
        domain.shutdown()?;
        Ok(())
    }

    /// Force stop, undefine, and clean up a VM
    pub fn destroy_vm(&self, name: &str) -> Result<()> {
        let domain = self.lookup(name)?;
        if domain.is_active()? {
            // Force stop — destroy() is the libvirt term for force-off
            let _ = domain.destroy();
        }
        domain.undefine()?;
        Ok(())
    }

    /// List all waybox-managed domains
    pub fn list_domains(&self) -> Result<Vec<DomainInfo>> {
        let domains = self.conn.list_all_domains(0)?;
        let mut infos = Vec::new();
        for domain in domains {
            let name = domain.get_name()?;
            let (state_raw, _) = domain.get_state()?;
            infos.push(DomainInfo {
                name,
                state: DomainState::from(state_raw as u32),
            });
        }
        Ok(infos)
    }

    /// Get state of a specific domain
    pub fn get_domain_state(&self, name: &str) -> Result<DomainState> {
        let domain = self.lookup(name)?;
        let (state_raw, _) = domain.get_state()?;
        Ok(DomainState::from(state_raw as u32))
    }

    /// Look up a domain by name, returning a friendly error if not found
    fn lookup(&self, name: &str) -> Result<Domain> {
        Domain::lookup_by_name(&self.conn, name).map_err(|_| WayboxError::VmNotFound {
            name: name.to_string(),
        })
    }
}
```

```rust
pub fn render_nixos_config(config: &WayboxConfig, password: &str) -> Result<RenderedNixosConfig> {
    let base = BaseTemplate {
        name: &config.name,
        password,
        system_packages: &config.system_packages,
    }
    .render()?;

    let waypipe = if !config.headless {
        Some(
            WaypipeTemplate {
                display_port: VSOCK_DISPLAY_PORT,
                control_port: VSOCK_CONTROL_PORT,
            }
            .render()?,
        )
    } else {
        None
    };

    let audio = if !config.headless {
        Some(AudioTemplate { audio_port: VSOCK_AUDIO_PORT }.render()?)
    } else {
        None
    };

    let venus = if !config.headless {
        Some(VenusTemplate.render()?)
    } else {
        None
    };

    let virtiofs = if !config.shared_folders.is_empty() {
        Some(
            VirtiofsTemplate {
                shared_folders: &config.shared_folders,
                readonly: config.share_readonly,
            }
            .render()?,
        )
    } else {
        None
    };

    let flatpak = if !config.flatpak_packages.is_empty() {
        Some(
            FlatpakTemplate {
                flatpak_packages: &config.flatpak_packages,
            }
            .render()?,
        )
    } else {
        None
    };

    Ok(RenderedNixosConfig {
        base,
        waypipe,
        audio,
        venus,
        virtiofs,
        flatpak,
    })
}

/// Generate the top-level configuration.nix that imports all modules
pub fn generate_configuration_nix(rendered: &RenderedNixosConfig) -> String {
    let mut imports = vec!["./base.nix".to_string()];
    if rendered.waypipe.is_some() {
        imports.push("./waypipe.nix".to_string());
    }
    if rendered.audio.is_some() {
        imports.push("./audio.nix".to_string());
    }
    if rendered.venus.is_some() {
        imports.push("./venus.nix".to_string());
    }
    if rendered.virtiofs.is_some() {
        imports.push("./virtiofs.nix".to_string());
    }
    if rendered.flatpak.is_some() {
        imports.push("./flatpak.nix".to_string());
    }

    let import_lines: String = imports
        .iter()
        .map(|p| format!("    {p}"))
        .collect::<Vec<_>>()
        .join("\n");

    format!(
        "{{ config, pkgs, ... }}:\n{{\n  imports = [\n{import_lines}\n  ];\n}}\n"
    )
}
```

```rust
/// Write rendered NixOS config files to a temp directory and build a qcow2 image
pub fn build_image(config: &WayboxConfig, rendered: &super::RenderedNixosConfig) -> Result<PathBuf> {
    let build_dir = std::env::temp_dir().join(format!("waybox-build-{}", config.name));
    std::fs::create_dir_all(&build_dir).map_err(|e| WayboxError::Io {
        context: format!("creating build dir {:?}", build_dir),
        source: e,
    })?;

    // Write all rendered config files
    write_file(&build_dir, "base.nix", &rendered.base)?;
    if let Some(ref content) = rendered.waypipe {
        write_file(&build_dir, "waypipe.nix", content)?;
    }
    if let Some(ref content) = rendered.audio {
        write_file(&build_dir, "audio.nix", content)?;
    }
    if let Some(ref content) = rendered.venus {
        write_file(&build_dir, "venus.nix", content)?;
    }
    if let Some(ref content) = rendered.virtiofs {
        write_file(&build_dir, "virtiofs.nix", content)?;
    }
    if let Some(ref content) = rendered.flatpak {
        write_file(&build_dir, "flatpak.nix", content)?;
    }

    // Write configuration.nix
    let configuration_nix = super::generate_configuration_nix(rendered);
    write_file(&build_dir, "configuration.nix", &configuration_nix)?;

    // Run nixos-generators
    let config_path = build_dir.join("configuration.nix");
    log::info!("Building NixOS image from {:?}", config_path);

    let output = Command::new("nixos-generate")
        .arg("--format")
        .arg("qcow")
        .arg("--configuration")
        .arg(&config_path)
        .output()
        .map_err(|e| WayboxError::Io {
            context: "running nixos-generate".to_string(),
            source: e,
        })?;

    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        // Clean up build dir on failure
        let _ = std::fs::remove_dir_all(&build_dir);
        return Err(WayboxError::ImageBuild(stderr.to_string()));
    }

    // nixos-generate outputs the image path on stdout
    let image_output = String::from_utf8_lossy(&output.stdout).trim().to_string();
    let built_image = PathBuf::from(&image_output);

    // Move image to final location
    let images_dir = WayboxConfig::images_dir();
    std::fs::create_dir_all(&images_dir).map_err(|e| WayboxError::Io {
        context: format!("creating images dir {:?}", images_dir),
        source: e,
    })?;
    let final_path = config.image_path();
    std::fs::copy(&built_image, &final_path).map_err(|e| WayboxError::Io {
        context: format!("copying image to {:?}", final_path),
        source: e,
    })?;

    // Clean up
    let _ = std::fs::remove_dir_all(&build_dir);
    let _ = std::fs::remove_file(&built_image);

    log::info!("Image built: {:?}", final_path);
    Ok(final_path)
}

    std::fs::write(&path, content).map_err(|e| WayboxError::Io {
        context: format!("writing {:?}", path),
        source: e,
    })
}
```

```rust
/// Launch an application in the guest by sending a command via the control port
pub fn launch_in_guest(vsock_cid: u32, command: &str) -> Result<()> {
    use std::os::unix::net::UnixStream;

    // Connect to the guest launcher daemon via vsock
    // vsock address: CID:port
    let output = Command::new("socat")
        .arg("-")
        .arg(format!("VSOCK-CONNECT:{vsock_cid}:{VSOCK_CONTROL_PORT}"))
        .stdin(Stdio::piped())
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
        .map_err(|e| WayboxError::Waypipe(format!("failed to connect to guest: {e}")))?;

    // Send the command
    let mut child = output;
    if let Some(ref mut stdin) = child.stdin {
        writeln!(stdin, "{command}")
            .map_err(|e| WayboxError::Waypipe(format!("failed to send command: {e}")))?;
    }
    drop(child.stdin.take()); // Close stdin to signal end of input

    let output = child
        .wait_with_output()
        .map_err(|e| WayboxError::Waypipe(format!("failed to read response: {e}")))?;

    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        return Err(WayboxError::Waypipe(format!(
            "guest launch failed: {stderr}"
        )));
    }

    log::info!("Launched '{command}' in guest CID {vsock_cid}");
    Ok(())
}

/// Query the guest for installed .desktop applications
pub fn list_guest_apps(vsock_cid: u32) -> Result<Vec<GuestApp>> {
    let output = Command::new("socat")
        .arg("-")
        .arg(format!("VSOCK-CONNECT:{vsock_cid}:{VSOCK_CONTROL_PORT}"))
        .stdin(Stdio::piped())
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
        .and_then(|mut child| {
            if let Some(ref mut stdin) = child.stdin {
                let _ = writeln!(stdin, "__list_apps__");
            }
            drop(child.stdin.take());
            child.wait_with_output()
        })
        .map_err(|e| WayboxError::Waypipe(format!("failed to query guest apps: {e}")))?;

    let stdout = String::from_utf8_lossy(&output.stdout);
    let apps = stdout
        .lines()
        .filter_map(|line| {
            let parts: Vec<&str> = line.splitn(3, '|').collect();
            if parts.len() >= 2 {
                Some(GuestApp {
```

```rust
/// Start the host-side audio bridge (receives PipeWire audio from guest via vsock)
pub fn start_bridge(vsock_cid: u32) -> Result<Child> {

    let child = Command::new("socat")
        .arg(format!("VSOCK-LISTEN:{VSOCK_AUDIO_PORT},reuseaddr,fork"))
        .arg(format!("UNIX-CONNECT:{pulse_socket}"))
        .stdout(Stdio::null())
        .stderr(Stdio::null())
        .spawn()
        .map_err(|e| WayboxError::Audio(format!("failed to start audio bridge: {e}")))?;

    log::info!("Audio bridge started on vsock port {VSOCK_AUDIO_PORT}");
    Ok(child)
}

/// Find the PulseAudio/PipeWire socket path
fn get_pulse_socket() -> String {
    if let Ok(runtime_dir) = std::env::var("XDG_RUNTIME_DIR") {
        let path = format!("{runtime_dir}/pulse/native");
        if std::path::Path::new(&path).exists() {
            return path;
        }
    }
    // Fallback
    let uid = unsafe { libc::getuid() };
    format!("/run/user/{uid}/pulse/native")
}
```

```rust
/// Check that all host prerequisites are available
pub fn check_prerequisites() -> Result<()> {
    nixos::image::check_prerequisites()?;

    // Check waypipe
    std::process::Command::new("which")
        .arg("waypipe")
        .output()
        .ok()
        .filter(|o| o.status.success())
        .ok_or_else(|| WayboxError::PrerequisiteNotFound {
            tool: "waypipe".to_string(),
            hint: "Install waypipe for Wayland forwarding".to_string(),
        })?;

    // Check vhost_vsock module
    let vsock_loaded = std::path::Path::new("/dev/vsock").exists()
        || std::process::Command::new("lsmod")
            .output()
            .map(|o| String::from_utf8_lossy(&o.stdout).contains("vhost_vsock"))
            .unwrap_or(false);

    if !vsock_loaded {
        return Err(WayboxError::PrerequisiteNotFound {
            tool: "vhost_vsock".to_string(),
            hint: "Load the kernel module: sudo modprobe vhost_vsock".to_string(),
        });
    }

    Ok(())
}

/// Full VM creation flow
pub fn create_vm(config: &WayboxConfig) -> Result<()> {
    log::info!("Creating VM '{}'", config.name);

    // 1. Check prerequisites
    check_prerequisites()?;

    // 2. Check for existing VM
    let config_path = config.config_path();
    if config_path.exists() {
        return Err(WayboxError::VmAlreadyExists(config.name.clone()));
    }

    // 3. Generate password and save credentials
    let password = passwords::generate_password(16);
    let mut store = PasswordStore::load()?;
    store.set(&config.name, &password);
    store.save()?;
    log::info!("Generated password for VM '{}'", config.name);

    // 4. Save config
    config.save()?;

    // 5. Render NixOS templates
    println!("Rendering NixOS configuration...");
    let rendered = nixos::render_nixos_config(config, &password)?;

    // 6. Build NixOS image
    println!("Building NixOS image (this may take a while)...");
    let image_path = nixos::image::build_image(config, &rendered)?;
    println!("Image built: {}", image_path.display());

    // 7. Generate domain XML and define VM
    let xml = generate_domain_xml(config);
    let conn = LibvirtConnection::new()?;
    conn.define_vm(&xml)?;
    log::info!("VM '{}' defined in libvirt", config.name);

    println!("VM '{}' created successfully!", config.name);
    println!("  vsock CID: {}", config.vsock_cid);
    println!("  Password:  {}", password);
    println!("\nStart with: waybox start {}", config.name);

    Ok(())
}

/// Start a VM and its display session
pub fn start_vm(name: &str) -> Result<()> {
    let config = WayboxConfig::load(name)?;

    let conn = LibvirtConnection::new()?;
    conn.start_vm(name)?;
    log::info!("VM '{}' started", name);

    if !config.headless {
        // Start display session
        let mut session = DisplaySession::new(name, config.vsock_cid);
        println!("Starting display session...");

        // Give the VM a moment to boot and start vsock services
        std::thread::sleep(std::time::Duration::from_secs(5));

        session.start()?;
        println!("VM '{}' is running. Launch apps with: waybox launch {} <command>", name, name);

        // Detach the session (processes continue in background)
        std::mem::forget(session);
    } else {
        println!("VM '{}' started (headless). Connect with: waybox console {}", name, name);
    }

    Ok(())
}

/// Stop a VM and clean up its display session
pub fn stop_vm(name: &str) -> Result<()> {
    let conn = LibvirtConnection::new()?;
    conn.stop_vm(name)?;
    println!("VM '{}' is shutting down", name);
    Ok(())
}

/// Destroy a VM: force stop, undefine, delete config and desktop files
pub fn destroy_vm(name: &str) -> Result<()> {
    // Remove from libvirt
    let conn = LibvirtConnection::new()?;
    match conn.destroy_vm(name) {
        Ok(_) => log::info!("VM '{}' removed from libvirt", name),
        Err(e) => log::warn!("Could not remove '{}' from libvirt: {}", name, e),
    }

    // Delete disk image
    let config = WayboxConfig::load(name).ok();
    if let Some(ref config) = config {
        let image = config.image_path();
        if image.exists() {
            std::fs::remove_file(&image).ok();
            log::info!("Deleted image {:?}", image);
        }
    }

    // Remove desktop files
    desktop::remove_desktop_files(name)?;

    // Remove password
    let mut store = PasswordStore::load()?;
    store.remove(name);
    store.save()?;

    // Delete config file (do this last)
    if let Some(config) = config {
        config.delete()?;
    }

    println!("VM '{}' destroyed", name);
    Ok(())
}
```
